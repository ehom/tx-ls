def display
  @dictionary.sort.to_h.each_pair do |key, value|
    puts "%32s:" % [ key ]
    display_pair("file-extensions", value["file-extensions"])
    display_pair("description", value["description"])
    display_pair("mimetype", value["mimetype"])
    puts
  end
end
